\section{Introduction to Vector Spaces}

\begin{definition}
    A \textbf{vector space} over a field $ \mathbb{F} $ is a set $ V $ equipped with two operations: vector addition and scalar multiplication, satisfying the eight vector space axioms.
\end{definition}

\begin{theorem}
    Let $ V $ be a vector space. The zero vector $ 0 \in V $ is unique.
\end{theorem}

\begin{proof}
    Suppose there exist two zero vectors, $ 0_1 $ and $ 0_2 $. By the property of additive identity, $ 0_1 + 0_2 = 0_1 $ (treating $ 0_2 $ as the identity) and $ 0_1 + 0_2 = 0_2 $ (treating $ 0_1 $ as the identity). Therefore, $ 0_1 = 0_2 $.
\end{proof}
